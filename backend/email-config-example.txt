# Email Configuration Example
# Copy these settings to your .env file

# Basic Email Configuration
EMAIL_PROVIDER=gmail
EMAIL_USER=your-email@gmail.com
EMAIL_APP_PASSWORD=your-gmail-app-password

# Company Information
COMPANY_EMAIL=contracts@cleaningpro.com
COMPANY_PHONE=+1 (555) 123-4567
COMPANY_ADDRESS=123 Business Avenue, Suite 100, City, State 12345

# Frontend URL (for email links)
FRONTEND_URL=http://localhost:4200

# How to set up Gmail App Password:
# 1. Go to your Google Account settings
# 2. Navigate to Security
# 3. Enable 2-Step Verification if not already enabled
# 4. Go to App passwords
# 5. Generate an app password for "Mail"
# 6. Use this app password in EMAIL_APP_PASSWORD (not your regular password)

# Alternative Providers:
# Outlook: EMAIL_PROVIDER=outlook, EMAIL_PASSWORD=your-password
# Custom SMTP: EMAIL_PROVIDER=smtp, SMTP_HOST=smtp.domain.com, etc.
# SendGrid: EMAIL_PROVIDER=sendgrid, SENDGRID_API_KEY=your-key
